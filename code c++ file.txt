
#include <ext/pb_ds/assoc_container.hpp> // Common file
#include <ext/pb_ds/tree_policy.hpp>
#include <functional> // for less
#include <iostream>
#include<bits/stdc++.h>
#define endl '\n'
#define ll long long int
#define pb push_back
#define pf push_front
const ll mod=1e9+7;
using namespace std;
ll ar[1000000];
vector<pair<ll,ll> >v1;
vector<pair<ll,ll> >v2;
vector<ll>v;


int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    ll i,j,t;
    ll p,a,n,b,c;
    string s;
    cin>>t;
    while(t--)
    {
        cin>>n;
        v.clear();
        v1.clear();
        v2.clear();
        for(i=0; i<n; i++)
        {
            cin>>a;
            v.pb(a);
        }
        cin>>s;
        for(i=0; i<n; i++)
        {
            if(s[i]=='0')
            {
                v1.pb({v[i],i+1});

            }
            else
                v2.pb({v[i],i+1});
        }
        sort(v1.begin(),v1.end());
        sort(v2.begin(),v2.end());
        p=1;
        for(i=0; i<v1.size(); i++)
        {
            a=v1[i].second;
            ar[a]=p;
            p++;
        }
        for(i=0; i<v2.size(); i++)
        {
            a=v2[i].second;
            ar[a]=p;
            p++;
        }
        for(i=1; i<=n; i++)
            cout<<ar[i]<<" ";
        cout<<endl;
    }

    return 0;


}
