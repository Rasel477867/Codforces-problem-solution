
#include <ext/pb_ds/assoc_container.hpp> // Common file

#include <functional> // for less

#define endl '\n'
#define ll long long int
#define pb push_back
#define pf push_front
const ll mod=1e9+7;
using namespace std;

deque<ll>v;
int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    ll t,i,j;
    ll ans,a,b,n,m,d,c,y;
    cin>>t;
    while(t--)
    {
        v.clear();
        cin>>m>>n;
        y=0;
        while(n!=0 || m!=0)
        {
            a=n%10;
            b=m%10;
            n=n/10;
            m=m/10;

            if(b>a)
            {
                d=n%10;
                a=d*10+a;
                n/=10;
            }
            c=a-b;

            if(c<=9 && c>=0)
                v.pb(c);
            else
            {
                y=1;
                break;
            }
        }

        if(y==0 && m==0 && n==0)
        {
            for(i=v.size()-1; i>=0; i--)
            {
                if(v.back()==0)
                    v.pop_back();
                else
                    break;
            }
            for(i=v.size()-1;i>=0; i--)
                cout<<v[i];
            cout<<endl;
        }
        else
            cout<<-1<<endl;
    }


return 0;

}
